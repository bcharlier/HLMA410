		width=0.45\textwidth,

		axis x line=middle,
		%y post scale=.24,
		%x post scale=.6,

		y label style={at={(current axis.above origin)},anchor= east},

		%yticklabels={0,0.1,0.2},
		xtick={0,10,20},
		%xticklabels={0,10,20},

\end{axis}
